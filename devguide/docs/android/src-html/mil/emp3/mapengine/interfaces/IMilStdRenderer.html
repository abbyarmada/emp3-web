001package mil.emp3.mapengine.interfaces;
002
003import android.util.SparseArray;
004
005import org.cmapi.primitives.IGeoBounds;
006
007import mil.emp3.api.MilStdSymbol;
008import mil.emp3.api.interfaces.IFeature;
009import mil.emp3.api.interfaces.core.IStorageManager;
010
011/**
012 *
013 * This interface defines access to the core MilStd rendering capability required by the map instances.
014 */
015public interface IMilStdRenderer {
016    void setStorageManager(IStorageManager storageManager);
017
018    /**
019     * This method sets the Cache Directory used by renderer and also sets the available heap memory for the application
020     * @param sCacheDir
021     * @param getMemoryClass
022     */
023    void setRendererCacheDir(String sCacheDir, int getMemoryClass);
024
025    /**
026     * This method is called by the map instances to get a list of renderable that compose the
027     * MilStd tactical graphic symbol.
028     * @param mapInstance
029     * @param symbol
030     * @param selected
031     * @return A list of basic IFeature objects. See {@link IFeature).
032     */
033    java.util.List<IFeature> getTGRenderableShapes(IMapInstance mapInstance, MilStdSymbol symbol, boolean selected);
034    
035    /**
036     * This method is called by the map instances the bitmap that compose the
037     * MilStd icon symbol.
038     * @param sSymbolCode The symbol code that identifies the 2525 symbol.
039     * @param oModifiers An array of modifiers to apply to the icon.
040     * @param oAttr An array of attributes to use for the icon.
041     * @return {@link IEmpImageInfo}.
042     */
043    IEmpImageInfo getMilStdIcon(String sSymbolCode, SparseArray oModifiers, SparseArray oAttr);
044
045    /**
046     * This method returns a SparseArray of the 2525 modifiers define in the symbol.
047     * @param mapInstance The map instance making the call.
048     * @param symbol the feature.
049     * @return SparseArray as per MilStd 2525 renderer
050     */
051    SparseArray<String> getUnitModifiers(IMapInstance mapInstance, MilStdSymbol symbol);
052
053    /**
054     * This method returns a SparseArray of the 2525 modifiers define in the tactical graphic symbol.
055     * @param mapInstance The map instance making the call.
056     * @param symbol the feature.
057     * @return SparseArray as per MilStd 2525 renderer
058     */
059    SparseArray<String> getTGModifiers(IMapInstance mapInstance, MilStdSymbol symbol);
060
061    /**
062     * This method returns a SparseArray of the attributes define in the symbol.
063     * @param mapInstance The map instance making the call.
064     * @param symbol the feature.
065     * @return SparseArray as per MilStd 2525 renderer
066     */
067    SparseArray<String> getAttributes(IMapInstance mapInstance, MilStdSymbol symbol, boolean selected);
068
069    /**
070     * This method retrieves the far distance threshold for the map. All single point MilStd icons at or
071     * beyond the far distances threshold (from the camera) are displayed as dots.
072     * @param mapInstance The map instance for request.
073     * @return The far distance threshold in meters.
074     */
075    double getFarDistanceThreshold(IMapInstance mapInstance);
076
077    /**
078     * This method retrieves the mid distance threshold for the map. All single point MiloStd icons at
079     * a distance less than the mid distance threshold are displayed as fully qualified MilStd icons. The
080     * labels displayed will depend on the maps MilStd label setting.
081     * the distances
082     * @param mapInstance The map instance for request.
083     * @return The mid distance threshold in meters.
084     */
085    double getMidDistanceThreshold(IMapInstance mapInstance);
086
087    /**
088     * Returns the name of the cache being used. In the current implementation if we want to use the RenderCache implemented
089     * by NASA WW then MilStdRenderer must use "NoBitmapCache".
090     * If we want to use the EMP cache then "BitmapCache" or "AdaptiveBitmapCache" should be used.
091     * "BitmapCache" and "AdaptiveBitmapCache" does not work well with NAS WW but is kept in place if we move to or add a new map engine.
092     * @return
093     */
094    String getBitmapCacheName();
095
096    /**
097     * this methos return the selected icon scale factor for the map instance.
098     * @param mapInstance
099     * @return
100     */
101    double getSelectedIconScale(IMapInstance mapInstance);
102}
