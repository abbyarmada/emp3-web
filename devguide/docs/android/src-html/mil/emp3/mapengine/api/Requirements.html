001package mil.emp3.mapengine.api;
002
003import mil.emp3.mapengine.interfaces.IMapEngineRequirements;
004
005/**
006 * This class implements the IMapEngineRequirements interface. A map engine
007 * developer is free to implement its own class as long as the interface is
008 * implemented.
009 */
010public class Requirements implements IMapEngineRequirements {
011    private final boolean wmsCapabilities;
012    
013    public Requirements(boolean requireWMSCapabilities) {
014        this.wmsCapabilities = requireWMSCapabilities;
015    }
016
017    @Override
018    public boolean requiresWMSCapabilities() {
019        return this.wmsCapabilities;
020    }
021}
