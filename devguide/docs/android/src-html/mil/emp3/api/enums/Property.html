001package mil.emp3.api.enums;
002
003/**
004 * This class enumerates all the configuration property keys used by properties parameter of different API calls. Each enumerated
005 * value requires a specific value type.
006 */
007public enum Property {
008    /**
009     * This enumerated value defines the map's fully qualified class name for a new map. The key value pair requires
010     * a string value.
011     */
012    ENGINE_CLASSNAME ("engine.className"),
013    /**
014     * This enumerated value defines the map's fully qualified android APK name where the class indicated
015     * in ENGINE_CLASSNAME is implemented. The key value pair requires a string value.
016     */
017    ENGINE_APKNAME   ("engine.apkName")  ,
018    /**
019     * This enumerated value defines the stoke color used to select features. The key value pair requires an IGeoColor
020     * value. The default color is yellow.
021     */
022    DEFAULT_SELECT_STROKE_COLOR ("default.stroke.color"),
023    /**
024     * This enumerated value defines the scale factor to use when an icon type feature is selected. The key value pair
025     * requires a double value. The default value is 1.2.
026     */
027    DEFAULT_SELECT_ICON_SCALE   ("default.icon.scale"),
028    /**
029     * This enumerated value defines the debug setting. The key value pair requires a boolean value and the default is
030     * false. Setting this value to true will generate additional log statements.
031     */
032    DEBUG                       ("debug"),
033    /**
034     * This enumerated value defines the proxy url to be used when a feature or map service indicates
035     * useProxy set to true. The default value is null. The key value pair requires a string value.
036     */
037    PROXY_URL                   ("proxy.url"),
038    /**
039     * This enumerated value defines if the mirror cache is to be used and in what mode. The default
040     * value is {@link mil.emp3.api.enums.MirrorCacheModeEnum#DISABLED}. This property requires a
041     * {@link mil.emp3.api.enums.MirrorCacheModeEnum} value.
042     */
043    MIRROR_CACHE_MODE           ("mirror.cache.mode"),
044    ;
045
046    final private String value;
047    Property(String value) {
048        this.value = value;
049    }
050
051    public String getValue() {
052        return value;
053    }
054
055    static public Property fromValue(String v) {
056        for (Property c : Property.values()) {
057            if (c.value.equals(v)) {
058                return c;
059            }
060        }
061        throw new IllegalArgumentException(v);
062    }
063}
