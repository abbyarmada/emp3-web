001package mil.emp3.api.enums;
002
003import mil.emp3.api.interfaces.IEventEnum;
004
005/**
006 * This class enumerates the type of feature draw events
007 */
008public enum FeatureDrawEventEnum implements IEventEnum {
009    /**
010     * This value indicates that the draw has started. The event target provides the feature being drawn.
011     */
012    DRAW_START,
013    /**
014     * This value indicates that the feature has been changed. The event target is a reference to the updated feature.
015     */
016    DRAW_UPDATE,
017    /**
018     * This value indicates that the draw has been canceled.
019     */
020    DRAW_CANCELED,
021    /**
022     * This value indicates that the draw has completed. The event target is a reference of the drawn feature.
023     */
024    DRAW_COMPLETE
025}
