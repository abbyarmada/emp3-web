001package mil.emp3.api.enums;
002
003import mil.emp3.api.interfaces.IEventEnum;
004
005/**
006 * This class enumerates the possible events that are generated during a freehand draw operation.
007 */
008public enum MapFreehandEventEnum implements IEventEnum{
009    /**
010     * This enumerated value indicates that the map has placed itself
011     * into freehand draw mode.
012     */
013    MAP_ENTERED_FREEHAND_DRAW_MODE,
014    /**
015     * This enumerated value indicates that the user started to draw on the map.
016     */
017    MAP_FREEHAND_LINE_DRAW_START,
018    /**
019     * This enumerated value indicates that the that the line has been updated.
020     */
021    MAP_FREEHAND_LINE_DRAW_UPDATE,
022    /**
023     * This enumerated value indicates that the user finished drawing a line (lifted its finger).
024     */
025    MAP_FREEHAND_LINE_DRAW_END,
026    /**
027     * This enumerated value indicates that the map has exited freehand draw mode.
028     */
029    MAP_EXIT_FREEHAND_DRAW_MODE
030}
