506     * This constructor creates MilStdSymbol, generates the GeoMilSymbol it encapsulates, and sets the
507     * MilStd version and symbol code to the values provided.
508     * @param eStandard {@link IGeoMilSymbol.SymbolStandard}
509     * @param sSymbolCode A valid MilStd symbol code.
510     * @throws mil.emp3.api.exceptions.EMP_Exception
511     */
512    public MilStdSymbol(IGeoMilSymbol.SymbolStandard eStandard, String sSymbolCode) throws EMP_Exception {
513        super(new GeoMilSymbol(), FeatureTypeEnum.GEO_MIL_SYMBOL);
514        
515        if ((sSymbolCode == null) || (sSymbolCode.length() < 15)) {
516            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "Invalid symbol code.");
517        }
518
519        if (eStandard == null) {
520            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "Invalid symbol standard.");
521        }
522
523        this.setSymbolStandard(eStandard);
524        this.setSymbolCode(sSymbolCode);
525        setStrokeStyle(null);
526        setFillStyle(null);
527        setLabelStyle(null);
528    }

530    /**
531     * This constructor creates MilStdSymbol and encapsulates the IGeoMilSymbol object provided.
532     * If the standard version is not set, it is set to the default value. If the modifiers are
533     * null it is set to an empty modifier list.
534     * @param oRenderable an object that implements the IGeoMilSymbol interface.
535     * @throws mil.emp3.api.exceptions.EMP_Exception This exception is raised if the geo renderable
536     * provided has an invalid symbol code.
537     */
538    public MilStdSymbol(IGeoMilSymbol oRenderable) throws EMP_Exception {
539        super(oRenderable, FeatureTypeEnum.GEO_MIL_SYMBOL);
540
541        if (this.getRenderable().getSymbolStandard() == null) {
542            this.setSymbolStandard(SymbolStandard.MIL_STD_2525C);
543        }
544
545        if (this.getRenderable().getModifiers() == null) {
546            this.getRenderable().setModifiers(new java.util.HashMap<IGeoMilSymbol.Modifier, String>());
547        }
548        
549        if ((oRenderable.getSymbolCode() == null) || (oRenderable.getSymbolCode().length() != 15)) {
550            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "The geo Renderable has an invalid symbol code.");
551        }
552
553        if (null == this.getAffiliation()) {
554            this.setAffiliation(Affiliation.FRIEND);
555        }
556    }

673    private java.util.ArrayList<String> convertToArrayList(String sCommaDelimitedString) {
674        java.util.ArrayList<String> aValueList = new java.util.ArrayList<>(java.util.Arrays.asList(sCommaDelimitedString.split("\\s*,\\s*")));
675
676        if (sCommaDelimitedString.isEmpty()) {
677            aValueList = new java.util.ArrayList<>();
678        } else {
679            aValueList = new java.util.ArrayList<>(java.util.Arrays.asList(sCommaDelimitedString.split("\\s*,\\s*")));
680        }
681
682        return aValueList;
683    }

685    /**
686     * This method inserts the numeric value of a comma delimited modifier at the specified index.
687     * @param eModifier See IGeoMilModifier.milModifierName. Not all modifier values are comma delimited values.
688     * @param iIndex The index of the value to insert. The first value is 0.
689     * @param fValue The numeric value to set. or Float.NaN to remove the value.
690     */
691    public void setModifier(IGeoMilSymbol.Modifier eModifier, int iIndex , float fValue) {
692        java.util.HashMap<IGeoMilSymbol.Modifier, String> oModifierList =
693                this.getRenderable().getModifiers();
694        String sValue = "";
695        java.util.ArrayList<String> aValueList;
696
697        if (oModifierList.containsKey(eModifier)) {
698            sValue = oModifierList.get(eModifier);
699        }
700
701        aValueList = this.convertToArrayList(sValue);
702
703        if ((iIndex >= 0) && (iIndex < aValueList.size())) {
704            if (Float.isNaN(fValue)) {
705                aValueList.remove(iIndex);
706            } else {
707                aValueList.add(iIndex, Float.toString(fValue));
708            }
709        } else if (!Float.isNaN(fValue)) {
710            // Add it to the end.
711            aValueList.add(Float.toString(fValue));
712        }
713
714        sValue = aValueList.toString(); // This operation encloses the string in "[]" we need to remove them.
715        if (sValue.length() > 0) {
716            sValue = sValue.substring(1, sValue.length() - 1);
717        }
718        this.setModifier(eModifier, sValue);
719    }

790    public boolean isSinglePoint() {
791        boolean ret = false;
792        String basicSymbolCode = this.getBasicSymbol();
793
794        if (SymbolUtilities.isTacticalGraphic(basicSymbolCode)) {
795            int milstdVersion = (this.getSymbolStandard() == IGeoMilSymbol.SymbolStandard.MIL_STD_2525B)? 0: 1;
796            SymbolDef symbolDefinition = SymbolDefTable.getInstance().getSymbolDef(basicSymbolCode, milstdVersion);
797
798            if (symbolDefinition.getDrawCategory() == SymbolDef.DRAW_CATEGORY_POINT) {
799                // This to account for TG that are icons.
800                ret = true;
801            }
802        } else {
803            ret = true;
804        }
805
806        return ret;
807    }
