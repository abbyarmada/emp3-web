001package mil.emp3.api;
002
003import org.cmapi.primitives.IGeoPosition;
004
005import java.util.List;
006
007import mil.emp3.api.enums.FeatureTypeEnum;
008import mil.emp3.api.abstracts.Feature;
009import mil.emp3.api.interfaces.IFeature;
010import org.cmapi.primitives.GeoRenderable;
011import org.cmapi.primitives.IGeoRenderable;
012
013/**
014 * This class implements the Path (Multi segment line) feature. It accepts a list of coordinates
015 * where each adjacent pair of coordinates identifies a line segment.
016 */
017public class Path extends Feature implements IFeature {
018    public Path() {
019        super(new GeoRenderable(), FeatureTypeEnum.GEO_PATH);
020    }
021
022    public Path(List<IGeoPosition> oCoordinates) {
023        super(new GeoRenderable(), FeatureTypeEnum.GEO_PATH);
024        this.setPositions(oCoordinates);
025    }
026
027    public Path(IGeoRenderable oRenderable) {
028        super(oRenderable, FeatureTypeEnum.GEO_PATH);
029    }
030}
