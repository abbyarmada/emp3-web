001package mil.emp3.api;
002
003import mil.emp3.api.interfaces.IContainer;
004import mil.emp3.api.interfaces.IEmpPropertyList;
005import mil.emp3.api.interfaces.IFeature;
006import mil.emp3.api.interfaces.IOverlay;
007import mil.emp3.api.interfaces.core.IStorageManager;
008import mil.emp3.api.utils.ManagerFactory;
009
010
011/**
012 *
013 * This static class provides global method to retrieve data or configuration setting.
014 */
015//  This class name is all lower case by design.
016public class global {
017    static final private IStorageManager storageManager = ManagerFactory.getInstance().getStorageManager();
018
019    /**
020     * This method finds a container with the specified ID.
021     * @param uuid The unique identifier of the container.
022     * @return IContainer or null if its not found.
023     */
024    public static IContainer findContainer(java.util.UUID uuid) {
025        return storageManager.findContainer(uuid);
026    }
027
028    /**
029     * This method finds an overlay with the specified ID.
030     * @param uuid The unique identifier of the overlay.
031     * @return IOveraly or null if its not found.
032     */
033    public static IOverlay findOverlay(java.util.UUID uuid) {
034        IContainer oContainer = global.findContainer(uuid);
035        
036        if (oContainer instanceof IOverlay) {
037            return (IOverlay) oContainer;
038        }
039        
040        return null;
041    }
042
043    /**
044     * This method finds a feature with the specified ID.
045     * @param uuid The unique identifier of the feature.
046     * @return A feature object or null if its not found.
047     */
048    public static IFeature findFeature(java.util.UUID uuid) {
049        IContainer oContainer = global.findContainer(uuid);
050        
051        if (oContainer instanceof IFeature) {
052            return (IFeature) oContainer;
053        }
054        
055        return null;
056    }
057
058    public static void setConfiguration(IEmpPropertyList properties) {
059
060    }
061}
