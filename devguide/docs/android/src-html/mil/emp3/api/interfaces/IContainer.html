001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoContainer;
004
005import mil.emp3.api.exceptions.EMP_Exception;
006import mil.emp3.api.listeners.EventListenerHandle;
007import mil.emp3.api.listeners.IContainerEventListener;
008import mil.emp3.api.listeners.IFeatureEditEventListener;
009import mil.emp3.api.listeners.IFeatureEventListener;
010import mil.emp3.api.listeners.IFeatureInteractionEventListener;
011import mil.emp3.api.listeners.IVisibilityEventListener;
012
013/**
014 * This interface defines the method for all containers.
015 */
016public interface IContainer extends IGeoContainer {
017    /**
018     * This method check if the container has children.
019     * @return True if it has one or more children, false otherwise.
020     */
021    public boolean hasChildren();
022    
023    /**
024     * This method returns a list of parents of this container.
025     * @return java.util.List of zero or more IContainers
026     */
027    public java.util.List<IContainer> getParents();
028
029    /**
030     * This method removes all children from this container.
031     * @throws EMP_Exception The exception is raised if a processing error is encountered.
032     */
033    public void clearContainer()
034            throws EMP_Exception;
035
036    /**
037     * This method registers a listener for all container events.
038     * The listener is called when any container event occurs on this container.
039     * @param listener The object that implements the proper listener for the container event.
040     * @return EventListenerHandle A handle to the registration which is needed to unregister the listener.
041     */
042    public EventListenerHandle addContainerEventListener(IContainerEventListener listener) throws EMP_Exception;
043
044    /**
045     * This method registers a listener for all visibility events.
046     * The listener is called when the visibility changes for any object in the container.
047     * @param listener The object that implements the proper listener for the visibility event.
048     * @return EventListenerHandle A handle to the registration which is needed to unregister the listener.
049     */
050    public EventListenerHandle addVisibilityEventListener(IVisibilityEventListener listener) throws EMP_Exception;
051
052    /**
053     * This method registers a listener for all feature events.
054     * The listener is called when any feature event occurs on this map.
055     * @param listener The object that implements the proper listener for the feature event.
056     * @return EventListenerHandle A handle to the registration which is needed to unregister the listener.
057     */
058    public EventListenerHandle addFeatureEventListener(IFeatureEventListener listener) throws EMP_Exception;
059
060    /**
061     * This method registers a listener for all user interaction events on all features in the container.
062     * The listener is called when any user interaction event occurs on any feature in this container.
063     * @param listener The object that implements the proper listener for the feature user interaction event.
064     * @return A handle to the registration which is needed to unregister the listener. See {@link EventListenerHandle}
065     */
066    public EventListenerHandle addFeatureInteractionEventListener(IFeatureInteractionEventListener listener) throws EMP_Exception;
067
068    /**
069     * This method registers a listener for all feature edit events for all features in the container.
070     * @param listener The object that implements the proper listener for the feature edit events.
071     * @return A handle to the registration which is needed to unregister the listener. See {@link EventListenerHandle}
072     */
073    public EventListenerHandle addFeatureEditEventListener(IFeatureEditEventListener listener) throws EMP_Exception;
074
075    /**
076     * This method removes the listener registration identified by the handle specified. The handle
077     * returned by the registration must be used to remove the registration to stop receiving events.
078     * all listeners must be removed before discarding the container.
079     * @param hListener The handle to the registration to remove.
080     */
081    public void removeEventListener(EventListenerHandle hListener);
082}
