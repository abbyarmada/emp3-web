001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoRenderable;
004
005import mil.emp3.api.enums.FeatureTypeEnum;
006import mil.emp3.api.exceptions.EMP_Exception;
007
008/**
009 * This interface is the base for all features.
010 */
011public interface IFeature extends IContainer, IGeoRenderable {
012
013    /**
014     * This method returns a reference of the GeoRenderable object encapsulated in the feature.
015     * @return See {@link IGeoRenderable}
016     */
017    public IGeoRenderable getRenderable();
018
019    /**
020     * This method returns the feature type.
021     * @return See {@link FeatureTypeEnum}
022     */
023    public FeatureTypeEnum getFeatureType();
024
025    /**
026     * This method retrieves a list of all the feature's child features.
027     * @return An ArrayList of zero or more IFeature objects.
028     */
029    public java.util.List<IFeature> getChildFeatures();
030
031    /**
032     * This method retrieves the list of all parent overlays of this feature.
033     * @return List of IOverlays
034     */
035    public java.util.List<IOverlay> getParentOverlays();
036
037    /**
038     * This method retrieves the list of parent features of this feature.
039     * @return List of IFeatures
040     */
041    public java.util.List<IFeature> getParentFeatures();
042
043    /**
044     * This method adds a child feature to this feature.
045     * @param feature The child to add.
046     * @param visible True if the feature is to be made visible or false otherwise.
047     * @throws EMP_Exception The exception is raised if a processing error is encountered.
048     */
049    public void addFeature(IFeature feature, boolean visible)
050            throws EMP_Exception;
051    /**
052     * This method adds one or more child features to this feature.
053     * @param features A list of features to add.
054     * @param visible True if the features are to be made visible or false otherwise.
055     * @throws EMP_Exception The exception is raised if a processing error is encountered.
056     */
057    public void addFeatures(java.util.List<IFeature> features, boolean visible)
058            throws EMP_Exception;
059
060    /**
061     * This method removes the child feature from this feature.
062     * @param feature The child to remove.
063     * @throws EMP_Exception The exception is raised if a processing error is encountered.
064     */
065    public void removeFeature(IFeature feature)
066            throws EMP_Exception;
067    /**
068     * This method removes one or more child feature from this feature.
069     * @param features A list of features to remove.
070     * @throws EMP_Exception The exception is raised if a processing error is encountered.
071     */
072    public void removeFeatures(java.util.List<IFeature> features)
073            throws EMP_Exception;
074
075    /**
076     * This method triggers an update of the feature. After changing any feature properties, this
077     * method MUST be called to force an update to be sent to the map(s).
078     */
079    public void apply();
080}
