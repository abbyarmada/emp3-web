001package mil.emp3.api.interfaces;
002
003import mil.emp3.api.exceptions.EMP_Exception;
004
005/**
006 * This interface class defines the interface to all Map Services such as
007 * WMS WMTS, etc ...
008 */
009public interface IMapService {
010    /**
011     * Set the name for this map service.
012     * @param sName the new name that will be set. A value of null sets the name 
013     * to an empty string.
014     */
015    public void setName(String sName);
016
017    /**
018     * This method retrieves the name property.
019     * @return String
020     */
021    public String getName();
022
023    /**
024     * This method retrieves the unique identifier for the service.
025     * @return java.util.UUID
026     */
027    public java.util.UUID getGeoId();
028
029    /**
030     * Set the name for this map service.
031     * @param sDescription the new name that will be set. A value of null sets the name 
032     * to an empty string.
033     */
034    public void setDescription(String sDescription);
035
036    /**
037     * This method retrieves the description property.
038     * @return String
039     */
040    public String getDescription();
041
042    /**
043     * This method retrieves the URL.
044     * @return java.net.URL
045     */
046    public java.net.URL getURL();
047
048    /**
049     * This method sets the layer list. The updated list is sent to all maps utilizing
050     * this service.
051     * @param newLayers A list of string. The list is ignored if its null or empty.
052     * @throws EMP_Exception 
053     */
054    public void setLayers(java.util.List<String> newLayers) throws EMP_Exception;
055
056    /**
057     * This method retrieves the current layer list.
058     * @return A list of String values.
059     */
060    public java.util.List<String> getLayers();
061}
