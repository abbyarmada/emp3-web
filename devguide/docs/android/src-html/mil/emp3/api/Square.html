001package mil.emp3.api;
002
003
004import org.cmapi.primitives.GeoSquare;
005import org.cmapi.primitives.IGeoPosition;
006import org.cmapi.primitives.IGeoSquare;
007
008import java.util.ArrayList;
009import java.util.List;
010
011import mil.emp3.api.abstracts.Feature;
012
013import mil.emp3.api.enums.FeatureTypeEnum;
014
015/**
016 * This class implements the EMP square feature. It accepts one (1) geographic coordinate that places the
017 * center of the square. The square's length of all sides is set with the setWidth method. And the square
018 * is centered at the coordinate.
019 */
020public class Square extends Feature implements IGeoSquare {
021
022    public Square() {
023        super(new GeoSquare(), FeatureTypeEnum.GEO_SQUARE);
024    }
025
026    public Square(IGeoPosition oCenter) {
027        super(new GeoSquare(), FeatureTypeEnum.GEO_SQUARE);
028        List<IGeoPosition> oCoordinates = new ArrayList<>();
029        oCoordinates.add(oCenter);
030        this.setPositions(oCoordinates);
031    }
032
033    public Square(IGeoPosition oCenter, double dWidth) {
034        super(new GeoSquare(), FeatureTypeEnum.GEO_SQUARE);
035        List<IGeoPosition> oCoordinates = new ArrayList<>();
036        oCoordinates.add(oCenter);
037        this.setPositions(oCoordinates);
038        this.setWidth(dWidth);
039    }
040
041    public Square(IGeoSquare oRenderable) {
042        super(oRenderable, FeatureTypeEnum.GEO_SQUARE);
043    }
044
045    /**
046     * This method returns a reference to the GeoSquare object encapsulated in the feature.
047     * @return See {@link IGeoSquare}
048     */
049    @Override
050    public final IGeoSquare getRenderable() {
051        return (IGeoSquare) super.getRenderable();
052    }
053
054    /**
055     * This method overrides the default width of a square feature.
056     * @param fValue The width in meters.
057     */
058    @Override
059    public void setWidth(double fValue) {
060        this.getRenderable().setWidth(fValue);
061    }
062
063    /**
064     * This method retrieves the width of the rectangle.
065     * @return The width in meters.
066     */
067    @Override
068    public double getWidth() {
069        return this.getRenderable().getWidth();
070    }
071}
