001package mil.emp3.api;
002
003import java.util.HashMap;
004import mil.emp3.api.abstracts.Feature;
005import mil.emp3.api.enums.FeatureTypeEnum;
006import mil.emp3.api.exceptions.EMP_Exception;
007import org.cmapi.primitives.GeoAirControlMeasure;
008import org.cmapi.primitives.IGeoAirControlMeasure;
009
010/**
011 *
012 * This class Implements an air control measure or airspace.
013 */
014public class AirControlMeasure extends Feature implements IGeoAirControlMeasure {
015
016    /**
017     * This constructor creates an ACM of the given type. If eACMType is null it generates an
018     * EMP_Exception.
019     * @param eAMCType A value enumerated value from AcmType
020     * @throws EMP_Exception If eACMType is null.
021     */
022    public AirControlMeasure(AcmType eAMCType) throws EMP_Exception {
023        super(new GeoAirControlMeasure(), FeatureTypeEnum.GEO_ACM);
024        
025        if (eAMCType == null) {
026            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "Invalid ACM Type.");
027        }
028        ((IGeoAirControlMeasure) super.getRenderable()).setAcmType(eAMCType);
029    }
030
031    /**
032     * This constructor creates an ACM from the given object.
033     * @param oGeoACM An object that implements the IGeoAirControlMeasure interface.
034     * @throws EMP_Exception If oGeoACM is null, or the object's AcmType is null.
035     */
036    public AirControlMeasure(IGeoAirControlMeasure oGeoACM) throws EMP_Exception {
037        super(oGeoACM, FeatureTypeEnum.GEO_ACM);
038        
039        if ((oGeoACM == null) || (oGeoACM.getAcmType() == null)) {
040            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "Parameter can not be null.");
041        }
042    }
043
044    @Override
045    public IGeoAirControlMeasure getRenderable() {
046        return (IGeoAirControlMeasure) super.getRenderable();
047    }
048    
049    @Override
050    public void setAcmType(AcmType at) {
051        this.getRenderable().setAcmType(at);
052    }
053
054    @Override
055    public AcmType getAcmType() {
056        return this.getRenderable().getAcmType();
057    }
058
059    @Override
060    public void setAttributes(HashMap<Attribute, String> hm) {
061        this.getRenderable().setAttributes(hm);
062    }
063
064    @Override
065    public HashMap<Attribute, String> getAttributes() {
066        return this.getRenderable().getAttributes();
067    }
068    
069}
