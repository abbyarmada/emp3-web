001package mil.emp3.api.events;
002
003import mil.emp3.api.interfaces.ICamera;
004import mil.emp3.api.interfaces.ILookAt;
005import mil.emp3.api.interfaces.IMap;
006import mil.emp3.api.enums.MapViewEventEnum;
007import org.cmapi.primitives.IGeoBounds;
008
009/**
010 * This event class is generated when a maps view is  changes.
011 */
012public abstract class MapViewChangeEvent extends Event<MapViewEventEnum, IMap> {
013    protected MapViewChangeEvent(MapViewEventEnum eventEnum, IMap oMap) {
014        super(eventEnum, oMap);
015    }
016
017    /**
018     * This method return the new view parameters.
019     * @return @See ICamera
020     */
021    public abstract ICamera getCamera();
022    
023    public abstract IGeoBounds getBounds();
024
025    public abstract ILookAt getLookAt();
026}
