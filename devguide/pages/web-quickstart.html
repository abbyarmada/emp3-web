Quick start for EMP3 Web

Below you will find a simple example to embed an use EMP3 in a web application.

Quick start HTML file

<!doctype html>
<html class="no-js" lang="">

<head>
    <meta charset="utf-8">
    <meta http-equiv="x-ua-compatible" content="ie=edge">
    <title></title>
    <meta name="description" content="">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <link rel="apple-touch-icon" href="apple-touch-icon.png">
    <!-- Place favicon.ico in the root directory -->
    <link href="vendor/mil-sym/renderer.css" rel="stylesheet" type="text/css" />
</head>

<body>
  <div style="position: absolute; top: 0px; left: 0px; right: 0px; bottom: 0px;" id="empmap"></div>

  <script src='vendor/jquery-1.11.2.min.js' type="text/javascript"></script>
  <script src='vendor/mil-sym/savm-bc.js' type="text/javascript"></script>
  <script src='emp3.min.js' type="text/javascript"></script>
  <script type="text/javascript">
    $(document).ready(function() {
        var map1 = new emp3.api.Map({
            container: "empmap",
            engine: {
    			engineBasePath: "emp3-cesium",
    			mapEngineId: "cesium"
    		},
            onSuccess: function() {
                console.log("Map creation success.");
            },
            onError: function(args) {
                console.log("Map creation failed.");
            }
        });
    });
  </script>
</body>
</html>
